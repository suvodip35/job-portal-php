<script>
    // Set up PDF.js worker
    pdfjsLib.GlobalWorkerOptions.workerSrc = 'https://cdnjs.cloudflare.com/ajax/libs/pdf.js/3.4.120/pdf.worker.min.js';

    let currentPdfFile = null;

    // DOM Elements
    const pdfInput = document.getElementById('pdfInput');
    const browseBtn = document.getElementById('browseBtn');
    const fileInfo = document.getElementById('fileInfo');
    const fileName = document.getElementById('fileName');
    const fileSize = document.getElementById('fileSize');
    const removeFile = document.getElementById('removeFile');
    const analyzeBtn = document.getElementById('analyzeBtn');
    const resultsSection = document.getElementById('resultsSection');
    const originalSize = document.getElementById('originalSize');
    const compressedSize = document.getElementById('compressedSize');
    const savingsAmount = document.getElementById('savingsAmount');
    const savingsPercentage = document.getElementById('savingsPercentage');

    // Initialize radio buttons and checkboxes
    function initializeControls() {
        // Radio buttons - custom implementation
        const radioButtons = document.querySelectorAll('input[type="radio"][name="compressionLevel"]');
        const radioLabels = document.querySelectorAll('.compression-option label');
        
        radioLabels.forEach(label => {
            label.addEventListener('click', function(e) {
                const radioId = this.getAttribute('for');
                const radio = document.getElementById(radioId);
                
                // Uncheck all radios
                radioButtons.forEach(r => r.checked = false);
                
                // Remove selected styles from all labels
                radioLabels.forEach(l => {
                    l.classList.remove('border-blue-500', 'bg-blue-50', 'dark:bg-blue-900/20');
                    const indicator = l.querySelector('.radio-indicator');
                    indicator.classList.remove('border-blue-500', 'bg-blue-500');
                    indicator.querySelector('span').classList.add('hidden');
                });
                
                // Check the clicked radio
                radio.checked = true;
                
                // Add selected styles to clicked label
                this.classList.add('border-blue-500', 'bg-blue-50', 'dark:bg-blue-900/20');
                const indicator = this.querySelector('.radio-indicator');
                indicator.classList.add('border-blue-500', 'bg-blue-500');
                indicator.querySelector('span').classList.remove('hidden');

                // If we have a file, update analysis
                if (currentPdfFile) {
                    updateCompressionAnalysis();
                }
            });
        });

        // Checkboxes
        const checkboxes = document.querySelectorAll('input[type="checkbox"]');
        checkboxes.forEach(checkbox => {
            checkbox.addEventListener('change', function() {
                // If we have a file, update analysis
                if (currentPdfFile) {
                    updateCompressionAnalysis();
                }
            });
        });

        // Set initial state for recommended option
        document.getElementById('level2').checked = true;
    }

    // Event Listeners
    browseBtn.addEventListener('click', () => pdfInput.click());
    pdfInput.addEventListener('change', handleFileSelect);
    removeFile.addEventListener('click', clearFile);
    analyzeBtn.addEventListener('click', generateCompressionReport);

    // Initialize controls on load
    document.addEventListener('DOMContentLoaded', initializeControls);

    // Drag and drop functionality
    const dropArea = document.querySelector('.border-dashed');
    ['dragenter', 'dragover', 'dragleave', 'drop'].forEach(eventName => {
        dropArea.addEventListener(eventName, preventDefaults, false);
    });

    function preventDefaults(e) {
        e.preventDefault();
        e.stopPropagation();
    }

    ['dragenter', 'dragover'].forEach(eventName => {
        dropArea.addEventListener(eventName, highlight, false);
    });

    ['dragleave', 'drop'].forEach(eventName => {
        dropArea.addEventListener(eventName, unhighlight, false);
    });

    function highlight() {
        dropArea.classList.add('border-blue-400', 'bg-blue-50', 'dark:bg-blue-900/20');
    }

    function unhighlight() {
        dropArea.classList.remove('border-blue-400', 'bg-blue-50', 'dark:bg-blue-900/20');
    }

    dropArea.addEventListener('drop', handleDrop, false);

    function handleDrop(e) {
        const dt = e.dataTransfer;
        const files = dt.files;
        if (files.length > 0 && files[0].type === 'application/pdf') {
            pdfInput.files = files;
            handleFileSelect();
        }
    }

    function handleFileSelect() {
        const file = pdfInput.files[0];
        if (!file) return;

        if (file.type !== 'application/pdf') {
            alert('Please select a PDF file.');
            return;
        }

        if (file.size > 50 * 1024 * 1024) {
            alert('File size must be less than 50MB.');
            return;
        }

        currentPdfFile = file;
        displayFileInfo(file);
        loadPdfPreview(file);
        updateCompressionAnalysis();
        resultsSection.classList.remove('hidden');
    }

    function displayFileInfo(file) {
        fileName.textContent = file.name;
        fileSize.textContent = formatFileSize(file.size);
        originalSize.textContent = formatFileSize(file.size);
        fileInfo.classList.remove('hidden');
    }

    function clearFile() {
        pdfInput.value = '';
        currentPdfFile = null;
        fileInfo.classList.add('hidden');
        resultsSection.classList.add('hidden');
        document.getElementById('originalPreview').innerHTML = '<p class="text-gray-500 dark:text-gray-400 text-center">PDF preview will appear here</p>';
    }

    async function loadPdfPreview(file) {
        try {
            const arrayBuffer = await file.arrayBuffer();
            
            // Try to load PDF for preview
            const pdf = await pdfjsLib.getDocument({ data: arrayBuffer }).promise;
            const page = await pdf.getPage(1);
            const viewport = page.getViewport({ scale: 0.5 });
            
            const canvas = document.createElement('canvas');
            const context = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;
            
            await page.render({
                canvasContext: context,
                viewport: viewport
            }).promise;
            
            document.getElementById('originalPreview').innerHTML = '';
            document.getElementById('originalPreview').appendChild(canvas);
        } catch (error) {
            console.error('Error loading PDF preview:', error);
            document.getElementById('originalPreview').innerHTML = 
                '<p class="text-red-500 text-center p-4">Error loading PDF preview. The file may be corrupted or encrypted.</p>';
        }
    }

    function updateCompressionAnalysis() {
        if (!currentPdfFile) return;

        // Get compression settings
        const compressionLevel = document.querySelector('input[name="compressionLevel"]:checked').value;
        const optimizeImages = document.getElementById('optimizeImages').checked;
        const removeMetadata = document.getElementById('removeMetadata').checked;
        const fontOptimization = document.getElementById('fontOptimization').checked;

        // Calculate compression ratio based on settings
        let compressionRatio = calculateCompressionRatio(compressionLevel, optimizeImages, removeMetadata, fontOptimization);
        
        const originalSizeBytes = currentPdfFile.size;
        const compressedSizeBytes = Math.floor(originalSizeBytes * compressionRatio);
        
        // Update UI
        updateResults(originalSizeBytes, compressedSizeBytes);
    }

    function calculateCompressionRatio(level, optimizeImages, removeMetadata, fontOptimization) {
        let ratio;
        
        // Base ratio based on compression level
        switch(level) {
            case 'extreme':
                ratio = 0.3; // 70% reduction
                break;
            case 'recommended':
                ratio = 0.6; // 40% reduction
                break;
            case 'light':
                ratio = 0.8; // 20% reduction
                break;
            default:
                ratio = 0.6;
        }

        // Apply additional compression based on options
        if (optimizeImages) {
            ratio *= 0.85; // 15% additional reduction
        }
        
        if (removeMetadata) {
            ratio *= 0.95; // 5% additional reduction
        }
        
        if (fontOptimization) {
            ratio *= 0.90; // 10% additional reduction
        }

        return ratio;
    }

    function updateResults(originalSizeBytes, compressedSizeBytes) {
        const originalFormatted = formatFileSize(originalSizeBytes);
        const compressedFormatted = formatFileSize(compressedSizeBytes);
        
        originalSize.textContent = originalFormatted;
        compressedSize.textContent = compressedFormatted;
        
        const savingsBytes = originalSizeBytes - compressedSizeBytes;
        const savingsPercent = ((savingsBytes / originalSizeBytes) * 100).toFixed(1);
        
        savingsAmount.textContent = `${formatFileSize(savingsBytes)} saved`;
        savingsPercentage.textContent = `${savingsPercent}% reduction`;
    }

    async function generateCompressionReport() {
        if (!currentPdfFile) return;

        // Get current settings
        const compressionLevel = document.querySelector('input[name="compressionLevel"]:checked').value;
        const optimizeImages = document.getElementById('optimizeImages').checked;
        const removeMetadata = document.getElementById('removeMetadata').checked;
        const fontOptimization = document.getElementById('fontOptimization').checked;

        const originalSizeBytes = currentPdfFile.size;
        const compressionRatio = calculateCompressionRatio(compressionLevel, optimizeImages, removeMetadata, fontOptimization);
        const compressedSizeBytes = Math.floor(originalSizeBytes * compressionRatio);
        const savingsPercent = ((1 - compressionRatio) * 100).toFixed(1);

        // Create a detailed report
        const reportContent = `
PDF Compression Analysis Report
===============================

File: ${currentPdfFile.name}
Original Size: ${formatFileSize(originalSizeBytes)}
Compression Level: ${compressionLevel.charAt(0).toUpperCase() + compressionLevel.slice(1)}

Compression Settings Applied:
- Image Optimization: ${optimizeImages ? 'Yes' : 'No'}
- Metadata Removal: ${removeMetadata ? 'Yes' : 'No'}
- Font Optimization: ${fontOptimization ? 'Yes' : 'No'}

Estimated Results:
- Compressed Size: ${formatFileSize(compressedSizeBytes)}
- Size Reduction: ${formatFileSize(originalSizeBytes - compressedSizeBytes)}
- Percentage Saved: ${savingsPercent}%

Note: This is an estimation based on compression algorithms.
Real PDF compression requires server-side processing with specialized tools.

Generated by FromCampus PDF Compression Simulator
        `.trim();

        // Create and download the report
        const blob = new Blob([reportContent], { type: 'text/plain' });
        const url = URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = `compression_report_${currentPdfFile.name.replace('.pdf', '')}.txt`;
        document.body.appendChild(a);
        a.click();
        document.body.removeChild(a);
        URL.revokeObjectURL(url);

        // Show success message
        alert('Compression report downloaded! Check your downloads folder.');
    }

    function formatFileSize(bytes) {
        if (bytes === 0) return '0 Bytes';
        
        const k = 1024;
        const sizes = ['Bytes', 'KB', 'MB', 'GB'];
        const i = Math.floor(Math.log(bytes) / Math.log(k));
        
        return parseFloat((bytes / Math.pow(k, i)).toFixed(2)) + ' ' + sizes[i];
    }
